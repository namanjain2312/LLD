3>Traverse a List in Java
  for (Integer num : numbers) {
      System.out.println(num);
  }

  for (int i = 0; i < numbers.size(); i++) {
      System.out.println(numbers.get(i));
  }

  Iterator<Integer> it = numbers.iterator();
  while (it.hasNext()) {
      System.out.println(it.next());
  }
